CREATE DATABASE  public_libraryDB;

use public_libraryDB;

CREATE TABLE authors (
                         author_id INT AUTO_INCREMENT PRIMARY KEY,
                         author_name VARCHAR(175) NOT NULL,
                         birth_year INT NOT NULL,
                         nationality VARCHAR(175) NOT NULL
);

CREATE TABLE publishers (
                            publisher_id INT AUTO_INCREMENT PRIMARY KEY,
                            publisher_name VARCHAR(175) NOT NULL,
                            location VARCHAR(175) NOT NULL
);

CREATE TABLE books (
                       book_id INT AUTO_INCREMENT PRIMARY KEY,
                       book_name VARCHAR(175) NOT NULL,
                       author_id INT,
                       year_published INT,
                       genre VARCHAR(175),
                       publisher_id INT,
                       page_number INT,
                       language VARCHAR(175),
                       original_language VARCHAR(175),
                       FOREIGN KEY (author_id) REFERENCES authors(author_id),
                       FOREIGN KEY (publisher_id) REFERENCES publishers(publisher_id)
);

CREATE TABLE members (
                         member_id INT AUTO_INCREMENT PRIMARY KEY,
                         member_name VARCHAR(175) NOT NULL,
                         birth_year INT NOT NULL,
                         sex VARCHAR(20) NOT NULL
);

CREATE TABLE staff (
                       staff_id INT AUTO_INCREMENT PRIMARY KEY,
                       staff_name VARCHAR(175) NOT NULL
);

CREATE TABLE transactions (
                              transaction_id INT AUTO_INCREMENT PRIMARY KEY,
                              member_id INT,
                              book_id INT,
                              staff_id INT,
                              transaction_type ENUM('Borrow', 'Return') NOT NULL,
                              transaction_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                              FOREIGN KEY (member_id) REFERENCES members(member_id),
                              FOREIGN KEY (book_id) REFERENCES books(book_id),
                              FOREIGN KEY (staff_id) REFERENCES staff(staff_id)
);

CREATE VIEW BookPublishedbyPublisher AS
SELECT
    publishers.publisher_name,
    COUNT(books.book_id) AS number_of_books
FROM
    books
        JOIN
    publishers ON books.publisher_id = publishers.publisher_id
GROUP BY
    publishers.publisher_name;


CREATE VIEW BobSmithTransactions AS
SELECT
    staff.staff_name,
    members.member_name,
    books.book_name,
    transactions.transaction_type
FROM
    transactions
        JOIN
    staff ON transactions.staff_id = staff.staff_id
        JOIN
    members ON transactions.member_id = members.member_id
        JOIN
    books ON transactions.book_id = books.book_id
WHERE
    staff.staff_name = 'Bob Smith';


CREATE VIEW BorrowsMichaelJhonson AS
SELECT
    members.member_name,
    books.book_name,
    transactions.transaction_type
FROM
    transactions
        JOIN
    members ON transactions.member_id = members.member_id
        JOIN
    books ON transactions.book_id = books.book_id
WHERE
    members.member_name = 'Michael Johnson' AND
    transactions.transaction_type = 'Borrow';

CREATE VIEW DetailedTransactions AS
SELECT
    transactions.transaction_id,
    members.member_name,
    books.book_name,
    staff.staff_name,
    transactions.transaction_type
FROM
    transactions
        JOIN
    members ON transactions.member_id = members.member_id
        JOIN
    books ON transactions.book_id = books.book_id
        JOIN
    staff ON transactions.staff_id = staff.staff_id;


CREATE VIEW DetailedBooks AS
SELECT
    books.book_name,
    authors.author_name,
    publishers.publisher_name
FROM
    books
        JOIN
    authors ON books.author_id = authors.author_id
        JOIN
    publishers ON books.publisher_id = publishers.publisher_id;
